    const ittenColors = [
      { name: "黃色 (一級)", rgb: [255, 235, 59], hex: "#FFEB3B" },
      { name: "黃橙 (三級)", rgb: [255, 193, 7], hex: "#FFC107" },
      { name: "橙色 (二級)", rgb: [255, 152, 0], hex: "#FF9800" },
      { name: "紅橙 (三級)", rgb: [255, 87, 34], hex: "#FF5722" },
      { name: "紅色 (一級)", rgb: [244, 67, 54], hex: "#F44336" },
      { name: "紅紫 (三級)", rgb: [233, 30, 99], hex: "#E91E63" },
      { name: "紫色 (二級)", rgb: [156, 39, 176], hex: "#9C27B0" },
      { name: "藍紫 (三級)", rgb: [103, 58, 183], hex: "#673AB7" },
      { name: "藍色 (一級)", rgb: [33, 150, 243], hex: "#2196F3" },
      { name: "藍綠 (三級)", rgb: [0, 188, 212], hex: "#00BCD4" },
      { name: "綠色 (二級)", rgb: [76, 175, 80], hex: "#4CAF50" },
      { name: "黃綠 (三級)", rgb: [139, 195, 74], hex: "#8BC34A" }
    ];

    const canvas = document.getElementById('wheel-canvas');
    const ctx = canvas.getContext('2d');
    const centerX = canvas.width / 2;
    const centerY = canvas.height / 2;
    const outerRadius = 180;
    const innerRadius = 70; // 中間挖空較大，更容易點擊外環
    
    // 儲存使用者選中的顏色索引 [color1_index, color2_index]
    let selectedIndices = [];

    // 繪製色環 (支援凸顯選中色)
    function drawWheel() {
      ctx.clearRect(0, 0, canvas.width, canvas.height);
      const arcSize = (2 * Math.PI) / 12;

      ittenColors.forEach((color, i) => {
        const startAngle = i * arcSize - arcSize / 2 - Math.PI / 2; // 將黃色轉到正上方
        const endAngle = (i + 1) * arcSize - arcSize / 2 - Math.PI / 2;
        
        ctx.beginPath();
        ctx.moveTo(centerX, centerY);
        
        // 如果被選中，稍微向外凸出並加上粗邊框
        let currentOuterRadius = outerRadius;
        if (selectedIndices.includes(i)) {
          currentOuterRadius = outerRadius + 15;
        }

        ctx.arc(centerX, centerY, currentOuterRadius, startAngle, endAngle);
        ctx.closePath();
        ctx.fillStyle = color.hex;
        ctx.fill();
        
        // 邊框繪製
        ctx.strokeStyle = selectedIndices.includes(i) ? "#2c3e50" : "#ffffff";
        ctx.lineWidth = selectedIndices.includes(i) ? 4 : 2;
        ctx.stroke();
      });

      // 挖空中間
      ctx.beginPath();
      ctx.arc(centerX, centerY, innerRadius, 0, 2 * Math.PI);
      ctx.fillStyle = "#ffffff";
      ctx.fill();
      ctx.strokeStyle = "#fff";
      ctx.stroke();
      
      // 在中間畫個小提示
      ctx.fillStyle = "#95a5a6";
      ctx.font = "14px sans-serif";
      ctx.textAlign = "center";
      ctx.fillText("點擊外圈", centerX, centerY - 8);
      ctx.fillText("選擇顏色", centerX, centerY + 12);
    }

    drawWheel();

    // 處理 Canvas 點擊事件 (計算極座標)
    canvas.addEventListener('click', function(event) {
      // 若正在旋轉中，禁止點擊
      if (document.getElementById('wheel-wrapper').classList.contains('spinning')) return;

      const rect = canvas.getBoundingClientRect();
      const x = event.clientX - rect.left - centerX;
      const y = event.clientY - rect.top - centerY;
      
      const distance = Math.sqrt(x * x + y * y);
      
      // 確保點擊在色環範圍內
      if (distance >= innerRadius && distance <= outerRadius + 15) {
        // 計算角度 (並校正起點，因繪圖時減了 Math.PI/2)
        let angle = Math.atan2(y, x) + Math.PI / 2;
        if (angle < 0) angle += 2 * Math.PI;
        
        // 將角度轉換為索引 (每個色塊佔 360/12 = 30度)
        const segmentAngle = (2 * Math.PI) / 12;
        let index = Math.round(angle / segmentAngle) % 12;
        
        handleColorSelection(index);
      }
    });

    // 處理選色邏輯
    function handleColorSelection(index) {
      // 隱藏之前的混色結果
      document.getElementById('result-box').style.display = 'none';

      const existPos = selectedIndices.indexOf(index);
      
      if (existPos !== -1) {
        // 如果點擊已選中的顏色，則取消選取
        selectedIndices.splice(existPos, 1);
      } else {
        if (selectedIndices.length < 2) {
          // 加入新顏色
          selectedIndices.push(index);
        } else {
          // 如果已經選了兩個，點第三個會替換掉第一個選的 (類似 FIFO)
          selectedIndices.shift();
          selectedIndices.push(index);
        }
      }

      updateSelectionUI();
      drawWheel();
    }

    // 更新右側選色狀態介面
    function updateSelectionUI() {
      const slot1 = document.getElementById('slot-1');
      const name1 = document.getElementById('name-1');
      const slot2 = document.getElementById('slot-2');
      const name2 = document.getElementById('name-2');
      const mixBtn = document.getElementById('mix-btn');

      // 更新第一個槽位
      if (selectedIndices.length >= 1) {
        const c1 = ittenColors[selectedIndices[0]];
        slot1.style.backgroundColor = c1.hex;
        slot1.classList.add('filled');
        name1.innerText = c1.name;
        name1.style.color = "#2c3e50";
        name1.style.fontWeight = "bold";
      } else {
        slot1.style.backgroundColor = "transparent";
        slot1.classList.remove('filled');
        name1.innerText = "尚未選擇";
        name1.style.color = "#7f8c8d";
        name1.style.fontWeight = "normal";
      }

      // 更新第二個槽位
      if (selectedIndices.length === 2) {
        const c2 = ittenColors[selectedIndices[1]];
        slot2.style.backgroundColor = c2.hex;
        slot2.classList.add('filled');
        name2.innerText = c2.name;
        name2.style.color = "#2c3e50";
        name2.style.fontWeight = "bold";
        
        // 啟用混色按鈕
        mixBtn.disabled = false;
        mixBtn.innerText = "🌀 開始旋轉混色！";
        mixBtn.style.animation = "pulse 1.5s infinite";
      } else {
        slot2.style.backgroundColor = "transparent";
        slot2.classList.remove('filled');
        name2.innerText = "尚未選擇";
        name2.style.color = "#7f8c8d";
        name2.style.fontWeight = "normal";
        
        // 禁用混色按鈕
        mixBtn.disabled = true;
        mixBtn.innerText = "🚫 請先選滿兩種顏色";
        mixBtn.style.animation = "none";
      }
    }

    // 重置選擇
    function resetSelection() {
      selectedIndices = [];
      updateSelectionUI();
      drawWheel();
      document.getElementById('result-box').style.display = 'none';
    }

    // 旋轉與混色
    function startMixing() {
      if (selectedIndices.length !== 2) return;

      const wrapper = document.getElementById('wheel-wrapper');
      wrapper.classList.add('spinning');
      document.getElementById('result-box').style.display = 'none';

      setTimeout(() => {
        wrapper.classList.remove('spinning');
        showResult();
      }, 1500);
    }

    function showResult() {
      const idx1 = selectedIndices[0];
      const idx2 = selectedIndices[1];
      const mode = document.getElementById('mode').value;

      const col1 = ittenColors[idx1].rgb;
      const col2 = ittenColors[idx2].rgb;

      let r, g, b;
      if (mode === 'equal') {
        r = Math.round((col1[0] + col2[0]) / 2);
        g = Math.round((col1[1] + col2[1]) / 2);
        b = Math.round((col1[2] + col2[2]) / 2);
      } else if (mode === 'heavy1') {
        r = Math.round((col1[0] * 2 + col2[0]) / 3);
        g = Math.round((col1[1] * 2 + col2[1]) / 3);
        b = Math.round((col1[2] * 2 + col2[2]) / 3);
      } else { // 多水渲染 (加入白色淡化)
        r = Math.round((col1[0] + col2[0] + 255 * 2) / 4);
        g = Math.round((col1[1] + col2[1] + 255 * 2) / 4);
        b = Math.round((col1[2] + col2[2] + 255 * 2) / 4);
      }

      const resHex = `rgb(${r}, ${g}, ${b})`;
      document.getElementById('result-swatch').style.backgroundColor = resHex;
      
      const n1 = ittenColors[idx1].name.split(' ')[0];
      const n2 = ittenColors[idx2].name.split(' ')[0];
      document.getElementById('result-text').innerText = `混色配方：${n1} + ${n2}`;
      
      // 判斷色彩關係
      const diff = Math.min(Math.abs(idx1 - idx2), 12 - Math.abs(idx1 - idx2));
      let relation = "這兩種顏色混合產生了豐富的水彩層次，你可以試著在畫紙上重現它！";
      
      if (diff === 6) {
        relation = "🌟 這是「互補色」混合！(在色環上成 180 度對立)。\n互補色相混會互相抵消彩度，調出沉穩的灰色或深灰褐色，是畫陰影和暗部的絕佳配方！";
      } else if (diff === 1) {
        relation = "🎨 這是「類似色」混合！(在色環上相鄰)。\n顏色非常協調自然，混色過程不易變髒，非常適合表現光影的平滑漸層。";
      } else if (diff === 4) {
         relation = "📐 這是「對比色」混合！(在色環上呈 120 度三角形)。\n色彩對比強烈，混合後會產生視覺衝擊力較強的中間色調。";
      }
      
      document.getElementById('result-desc').innerText = relation;
      document.getElementById('result-box').style.display = 'block';
      
      // 平滑捲動到底部觀看結果
      document.getElementById('result-box').scrollIntoView({ behavior: 'smooth', block: 'nearest' });
    }

    // 加入脈衝動畫樣式 (給按鈕使用)
    const style = document.createElement('style');
    style.innerHTML = `
      @keyframes pulse {
        0% { box-shadow: 0 0 0 0 rgba(230, 126, 34, 0.7); }
        70% { box-shadow: 0 0 0 15px rgba(230, 126, 34, 0); }
        100% { box-shadow: 0 0 0 0 rgba(230, 126, 34, 0); }
      }
    `;
    document.head.appendChild(style);

    // 視圖切換 (與之前相同)
    let isAdvanced = false;

      document.getElementById('basic-left').style.display = isAdvanced ? 'none' : 'flex';
      document.getElementById('basic-right').style.display = isAdvanced ? 'none' : 'flex';

    }

    // --- 進階模式相機邏輯保持不變 ---
    const video = document.getElementById('video-feed');
    const photoCanvas = document.getElementById('photo-canvas');
    const photoCtx = photoCanvas.getContext('2d');

    photoCtx.fillStyle = "#333";
    photoCtx.fillRect(0, 0, photoCanvas.width, photoCanvas.height);
    photoCtx.fillStyle = "#fff";
    photoCtx.font = "20px sans-serif";
    photoCtx.fillText("請點擊上方按鈕開啟相機或上傳照片", 130, 200);

    function startCamera() {
      video.style.display = 'block';
      photoCanvas.style.display = 'none';
      navigator.mediaDevices.getUserMedia({ video: true })
        .then(stream => { video.srcObject = stream; })
        .catch(err => { alert("無法開啟相機，請確認權限或使用圖片上傳！"); });
    }

    function takeSnapshot() {
      if (video.style.display !== 'none') {
        photoCanvas.style.display = 'block';
        photoCtx.drawImage(video, 0, 0, photoCanvas.width, photoCanvas.height);
        video.style.display = 'none';
        const stream = video.srcObject;
        if (stream) stream.getTracks().forEach(track => track.stop());
      }
    }

    document.getElementById('file-input').addEventListener('change', function(e) {
      const file = e.target.files[0];
      if (file) {
        const reader = new FileReader();
        reader.onload = function(event) {
          const img = new Image();
          img.onload = function() {
            photoCanvas.style.display = 'block';
            video.style.display = 'none';
            // 保持比例繪製
            const ratio = Math.min(photoCanvas.width / img.width, photoCanvas.height / img.height);
            const w = img.width * ratio;
            const h = img.height * ratio;
            const x = (photoCanvas.width - w) / 2;
            const y = (photoCanvas.height - h) / 2;
            photoCtx.clearRect(0,0,photoCanvas.width,photoCanvas.height);
            photoCtx.drawImage(img, x, y, w, h);
          }
          img.src = event.target.result;
        }
        reader.readAsDataURL(file);
      }
    });

    photoCanvas.addEventListener('click', function(e) {
      if (!isAdvanced) return;
      const rect = photoCanvas.getBoundingClientRect();
      const x = (e.clientX - rect.left) * (photoCanvas.width / rect.width);
      const y = (e.clientY - rect.top) * (photoCanvas.height / rect.height);
      const pixel = photoCtx.getImageData(x, y, 1, 1).data;
      analyzePixelColor(pixel[0], pixel[1], pixel[2]);
    });

    function analyzePixelColor(r, g, b) {
      let bestMatch = "";
      let minDistance = Infinity;

      for (let i = 0; i < ittenColors.length; i++) {
        for (let j = i; j < ittenColors.length; j++) {
          const c1 = ittenColors[i].rgb;
          const c2 = ittenColors[j].rgb;
          const mixR = (c1[0] + c2[0]) / 2;
          const mixG = (c1[1] + c2[1]) / 2;
          const mixB = (c1[2] + c2[2]) / 2;

          const dist = Math.sqrt(Math.pow(r - mixR, 2) + Math.pow(g - mixG, 2) + Math.pow(b - mixB, 2));
          if (dist < minDistance) {
            minDistance = dist;
            if (i === j) {
              bestMatch = `極度接近伊登色環中的 **${ittenColors[i].name}**！`;
            } else {
              bestMatch = `這可以嘗試用 **${ittenColors[i].name.split(' ')[0]}** 與 **${ittenColors[j].name.split(' ')[0]}** 約 1:1 調配出來！`;
            }
          }
        }
      }
      
      document.getElementById('analysis-text').innerHTML = 
        `<div style="display:inline-block;width:30px;height:30px;background:rgb(${r},${g},${b});vertical-align:middle;border:2px solid #ccc;border-radius:5px;margin-right:15px;box-shadow:0 2px 5px rgba(0,0,0,0.2);"></div>` +
        `<span style="color:#2c3e50;font-size:20px;">💡 ${bestMatch}</span>`;
    }
  </script>
